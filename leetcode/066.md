    public static int[] plusOne(int[] digits) {
      for (int i = digits.length - 1; i > -1; i--) {
        int b = digits[i];
        b += 1;
        if (b % 10 == 0) {
          //说明进位了
          digits[i] = 0;
        } else {
          //没有进位就结束了
          digits[i] = b;
          return digits;
        }
      }
      int[] c=new int[digits.length+1];
      c[0]=1;
      return c;
    }
  
  }
